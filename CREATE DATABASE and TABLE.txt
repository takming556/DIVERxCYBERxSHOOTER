CREATE DATABASE dcs;
CREATE TABLE ranking_test(
	id INT AUTO_INCREMENT,
	nickname VARCHAR(16) COLLATE utf8mb4_bin NOT NULL,
	score INT UNSIGNED NOT NULL,
	device VARCHAR(8),
	timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
	PRIMARY KEY (id)
);

CREATE TABLE ranking(
	id INT AUTO_INCREMENT,
	nickname VARCHAR(16) COLLATE utf8mb4_bin NOT NULL,
	score INT UNSIGNED NOT NULL,
	device VARCHAR(8),
	timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
	PRIMARY KEY (id)
);
